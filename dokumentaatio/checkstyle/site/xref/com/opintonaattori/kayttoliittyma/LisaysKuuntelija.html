13  /**
14   * Luokka tarjoaa käyttöliittymän lisäyspainikkeen toiminnallisuuteen liittyvät metodit.
15   */
16  public class LisaysKuuntelija implements ActionListener {
17  
18      private Kayttaja kayttaja;
19      private JTextField lahde1;
20      private JTextField lahde2;
21      private JTextField lahde3;
22      private Kayttoliittyma kayttoliittyma;
23      
24      public LisaysKuuntelija(Kayttaja kayttaja, Kayttoliittyma kayttoliittyma, JTextField lahde1, JTextField lahde2, JTextField lahde3) {
25          this.kayttaja = kayttaja;
26          this.lahde1 = lahde1;
27          this.lahde2 = lahde2;
28          this.lahde3 = lahde3;
29          this.kayttoliittyma = kayttoliittyma;
30      }
31      
32      /**
33       * Metodi lisää käyttäjälle uuden kurssisuorituksen 
34       * kun lisää nappia painetaan.
35       *
36       * @param e annettu ActionEvent
37       *
38       * @see com.opintonaattori.logiikka.Kayttaja#lisaaKurssisuoritus(java.lang.String, int, int)
39       */
40      @Override
41      public void actionPerformed(ActionEvent e) {
42          try {
43              this.kayttaja.lisaaKurssisuoritus(this.lahde1.getText(), intMuunnin(this.lahde2), intMuunnin(this.lahde3));
44              //this.kayttoliittyma.lisaaKurssi(this.kayttoliittyma.getFrame().getContentPane(), this.kayttoliittyma.getFrame().getContentPane().getLayout());
45              this.kayttoliittyma.paivita(this.kayttoliittyma.getFrame().getContentPane());
46              this.kayttoliittyma.runAgain(50);
47          } catch (IOException | NumberFormatException ex) {
48              Logger.getLogger(LisaysKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
49          }
50      }
51      
52      /**
53       * Metodi muuttaa saadun tekstikentän arvon kokonaisluvuksi.
54       *
55       * @param a muunnettavaksi haluttu JTextField
56       * @return annettu JTextFiel:n saama arvo kokonaislukuna
57       *
58       */
59      public int intMuunnin(JTextField a) {
60          int numero = Integer.parseInt(a.getText());
61          return numero;
62      }
63      
64  }
