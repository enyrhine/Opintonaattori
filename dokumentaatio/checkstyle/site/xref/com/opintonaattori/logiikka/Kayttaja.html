22      public Kayttaja(String nimi) throws IOException {
23          this.nimi = nimi;
24          this.kurssisuoritukset = new ArrayList<>();
25          this.tiedosto = new File("src/resources/" + this.nimi + ".csv");
26          this.tallentaja = new Tallentaja(this.tiedosto);
27          this.raportti = new Raporttikone(this.tiedosto);
28      }
29  
30      /**
31       * Metodi lisää käyttäjälle uuden kurssisuorituksen arvosanoineen ja
32       * opintopisteineen.
33       *
34       * @param nimi Käyttäjän antama kurssin nimi
35       * @param op Käyttäjän antama opintopistemäärä
36       * @param arvosana Käyttäjän antama arvosana
37       *
38       * @see
39       * com.opintonaattori.tiedosto.Tallentaja#lisaaKurssisuoritus(com.opintonaattori.logiikka.Kurssisuoritus)
40       */
41      public void lisaaKurssisuoritus(String nimi, int op, int arvosana) throws IOException {
42          Kurssisuoritus kurssi = new Kurssisuoritus(nimi, op, arvosana);
43         // this.kurssisuoritukset.add(kurssi);
44          this.tallentaja.lisaaKurssisuoritus(kurssi);
45      }

57      /**
58       * Metodi hakee kaikki tiedostossa olevat kurssit ja palauttaa ne listana jolla Stringinä kurssin tiedot.
59       *
60       * @see 
61       * com.opintonaattori.tiedosto.Raporttikone#lueTiedosto() 
62       * @see 
63       * com.opintonaattori.tiedosto.Raporttikone#lueRivit(int) 
64       * 
65       * @return ArrayListin kursseista
66       *
67       */
68      public ArrayList<String> getKurssit() {
69          ArrayList<String> kaikkiKurssit = new ArrayList<>();
70          for (int i = 0; i < this.raportti.lueTiedosto().size(); i++) {
71              String[] kurssiTiedot = this.raportti.lueRivit(i);
72              kaikkiKurssit.add(kurssiTiedot[0] + " " + kurssiTiedot[1] + " " + kurssiTiedot[2]);
73          }
74          return kaikkiKurssit;
75      }
76      
77      public String tulostaKurssit(int i) {
78          return getKurssit().get(i);
79      }
80      
81      public String tulostaKurssinNimi(int i) {
82          return getKurssi(i)[0];
83      }
84      
85      public String tulostaKurssinOpintopisteet(int i) {
86          return getKurssi(i)[1];
87      }
88      
89      public String tulostaKurssinArvosana(int i) {
90          return getKurssi(i)[2];
91      }
92  
93      public File getTiedosto() {
94          return this.tiedosto;
95      }
96      
97      public String[] getKurssi(int i) {
98          return this.raportti.lueRivit(i);
99      }
