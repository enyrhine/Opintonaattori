95      /**
96       * Metodi luo uuden lisayskuuntelijan ja asettaa sen arvoiksi halutut
97       * kurssin parametrit.
98       *
99       * @param kurssinNimi annettu JTextField
100      * @param op annettu JTextField
101      * @param arvosana annettu JTextField
102      * @param lisaa annettu JButton
103      *
104      * @see
105      * com.opintonaattori.kayttoliittyma.LisaysKuuntelija#actionPerformed(java.awt.event.ActionEvent)
106      */
107     public LisaysKuuntelija setLisayskuuntelija(JTextField kurssinNimi, JTextField op, JTextField arvosana, JButton lisaa) {
108         LisaysKuuntelija lisaysKuuntelija = new LisaysKuuntelija(this.kayttaja, kurssinNimi, op, arvosana);
109         lisaa.addActionListener(lisaysKuuntelija);
110         ActionEvent tapahtuma = new ActionEvent(lisaysKuuntelija, 2, "lisää");
111         lisaysKuuntelija.actionPerformed(tapahtuma);
112         return lisaysKuuntelija;
113     }
